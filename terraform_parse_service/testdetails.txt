Create terraform parse service
Please Create a backend service to render the terraform file and create a new branch & PR to github repo.
Service which can receive rest api request ( POST method ) with certain payload.
Parse this payload and render to terraform file for create S3 bucket.
File example:
{
  "payload":{
    "properties":{
       "aws-region":"eu-west-1",
       "acl":"private",
       "bucket-name": "tripla-bucket"
       }
    }
}
provider "aws" {
    alias = "bucket_region"
    region = "eu-west-1"
  }
  
resource "aws_s3_bucket" "example" {
  provider = aws.bucket_region
  bucket = "tripla-bucket"
}

resource "aws_s3_bucket_acl" "example_bucket_acl" {
   bucket = aws_s3_bucket.example.id
   acl = "private"
}
